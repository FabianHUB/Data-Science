import numpy as np
import pandas as pd

df = pd.read_csv("telco.csv")

feature_name =  str(input('Waiting for input: '))  # example: Age
num_bins = int(float(input('Waiting for input: '))) # example: 5


def equal_width_binning(feature_name, num_bins, df=df):
    
    # Inputs:
    # feature_name: a string of a feature name in df
    # num_bins: a positive integer greater than 1
    # df: A pandas dataframe

    # Outputs:
    # bins: a numpy array
    
    # Get the values of the feature
    feature = df[feature_name]
    
    ### ADD YOUR CODE BELOW
        
    
    # Get the range of the feature
    feature_min = min(feature)
    feature_max = max(feature)
    
    # Calculate equal bin widths 
    bin_ranges = np.linspace(feature_min, feature_max, num_bins+1)
        
    # Initialise a numpy array to the largest bin (don't need a seperate condition for the maximum entry)
    bins = np.repeat(num_bins, len(feature))
    
    # Assign bins
    for index in range(1, num_bins+1):
        bins[(feature >= bin_ranges[index-1]) & (feature < bin_ranges[index])] = index
        
    # bins = np.repeat(999, len(feature))
    
    ### ADD YOUR CODE ABOVE

    return bins


# we will print the first 10 bin membership for simplicity
print(f"{equal_width_binning(feature_name, num_bins)[0:10]}")
